预编译发生在函数执行的前一刻：

  1. 创建AO对象(Activation Object)（执行期上下文）

// 求一个字符串的字节长度
// 当前字符位的unicode > 255, 那么该字符字节长度为2；<= 255, 为 1
var str = "hello world 风之子";
// str.charCodeAt(0)
function bytesLength(str) {
  var count = str.length;
  for (var i = 0; i < str.length; i++) {
    if (str.charCodeAt(i) > 255) {
      count++;
    }
  }
  // var count = 0;
  // for (var i = 0; i < str.length; i++) {
  //   if (str.charCodeAt(i) > 255) {
  //     count += 2;
  //   } else {
  //     count++;
  //   }
  // }
  return count;
}
bytesLength(str);
