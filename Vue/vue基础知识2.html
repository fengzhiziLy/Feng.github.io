# 给组件绑定原生事件

<body>
  <div id="root">
    <child @click="handleClick"></child>
  </div>
</body>
<script type="text/javascript">
  Vue.component('child', {
    template: '<div @click="handleChildClick">Child</div>',
    methods: {
      handleChildClick: function () {
        alert('child click')
        this.$emit('click')
      }
    }
  })

  var app = new Vue({
    el: "#root",
    methods: {
      handleClick: function () {
        alert('click')
      }
    }
  })
</script>

<body>
  <div id="root">
    <child @click.native="handleClick"></child>  
    <!-- 注意上面的native -->
  </div>
</body>
<script type="text/javascript">
  Vue.component('child', {
    template: '<div>Child</div>'
  })

  var app = new Vue({
    el: "#root",
    methods: {
      handleClick: function () {
        alert('click')
      }
    }
  })
</script>

# 非父子组件间的传值

两中解决办法：

  1. Vuex---以后说

  2. 发布订阅模式/观察者模式/总线/Bus

<body>
  <div id="root">
    <child content="fengzi"></child>
    <child content="fengzhizi"></child>
  </div>
</body>
<script type="text/javascript">
  Vue.prototype.bus = new Vue()
  Vue.component('child', {
    data: function () {
      return {
        selfContent: this.content
      }
    },
    props: {
      content: String
    },
    template: '<div @click="handleClick">{{selfContent}}</div>',
    methods: {
      handleClick: function () {
        this.bus.$emit('change', this.selfContent)
      }
    },
    mounted: function () {
      var this_ = this
      this.bus.$on('change', function (msg) {
        this_.selfContent = msg
      })
    }
  })

  var app = new Vue({
    el: "#root"
  })
</script>

# 在vue中使用插槽

<body>
  <div id="root">
    <child content="<p>fengzhizi</p>"></child>
  </div>
</body>
<script type="text/javascript">
  Vue.component('child', {
    props: ['content'],
    template: `<div>
                <p>hello</p>
                <div v-html="this.content"></div>
              </div>`
  })
  // 这种情况下，在页面中能显示出来，但是实际上显示的是<div><p>hello</p></div>
  // 模版占位符template也是这样的效果
  var app = new Vue({
    el: "#root"
  })
</script>

<body>
  <div id="root">
    <child>
      <h1>fengzhizi</h1>
    </child>
  </div>
</body>
<script type="text/javascript">
  Vue.component('child', {
    template: `<div>
                <p>hello</p>
                <slot>默认内容</slot>
              </div>`
  })
  // 这样在页面中就能正确的显示出来
  var app = new Vue({
    el: "#root"
  })
</script>

<body>
  <div id="root">
    <body-content>
      <div class="header" slot="header">header</div>
      <div class="footer" slot="footer">footer</div>
    </body-content>
  </div>
</body>
<script type="text/javascript">
  Vue.component('body-content', {
    template: `<div>
                <slot name='header'></slot>
                <div class="content">content</div>
                <slot name='footer'></slot>
              </div>`
  })
  // 具名插槽
  var app = new Vue({
    el: "#root"
  })
</script>

# 作用域插槽

<body>
  <div id="root">
    <child>
      <template slot-scope="props">
        <h1>{{props.item}} - fengzhizi</h1>
      </template>
    </child>
  </div>
</body>
<script type="text/javascript">
  Vue.component('child', {
    data: function () {
      return {
        list: [1, 2, 3, 4]
      }
    },
    template: `<div>
              <ul>
                <slot v-for="item of list" :item=item></slot>
              </ul>
            </div>`
  })
  // 当子组件做循环或者某一部分的DOM应该由外部传递进来的时候，需要用作用域插槽
  var app = new Vue({
    el: "#root"
  })
</script>

# 动态组件与v-once指令

<body>
  <div id="root">
    <!-- <child-one v-if="type === 'child-one' "></child-one>
    <child-two v-if="type === 'child-two' "></child-two> -->
    <component :is="type"></component>
    <button @click="handleBtnClick">change</button>
  </div>
</body>
<script type="text/javascript">
  Vue.component('child-one', {
    template: '<div>child-one</div>'
  })
  Vue.component('child-two', {
    template: '<div>child-two</div>'
  })

  var app = new Vue({
    el: "#root",
    data: {
      type: 'child-one'
    },
    methods: {
      handleBtnClick: function () {
        this.type = this.type === 'child-one' ? 'child-two' : 'child-one'
      }
    }
  })
</script>

<body>
  <div id="root">
    <child-one v-if="type === 'child-one' "></child-one>
    <child-two v-if="type === 'child-two' "></child-two>
    <!-- <component :is="type"></component> -->
    <button @click="handleBtnClick">change</button>
  </div>
</body>
<script type="text/javascript">
  Vue.component('child-one', {
    template: '<div v-once>child-one</div>'
  })
  Vue.component('child-two', {
    template: '<div v-once>child-two</div>'
  })
  // v-once指令可以有效的提高静态内容的展示效率
  var app = new Vue({
    el: "#root",
    data: {
      type: 'child-one'
    },
    methods: {
      handleBtnClick: function () {
        this.type = this.type === 'child-one' ? 'child-two' : 'child-one'
      }
    }
  })
</script>
