# 组件化

# 组件的封装

class Todo extends Component {
  constructor (props) {
    super(props)
    this.state = {
      list: []    // 组件的数据
    }
  }
  render () {
    return (  // 组件的视图
      <div>
        <Input addTitle={this.addTitle.bind(this)} />
        <List data={this.state.list} />
      </div>
    )
  }
  addTitle (title) {
    const currentList = this.state.list
    this.setState({   // 组件的变化逻辑
      list: currentList.concat(title)
    })
  }
}

# 组件的复用

import List from './list/index'
import Input from './input/index'

class Todo extends Component {
  constructor (props) {
    super(props)
    this.state = {
      list: []
    }
  }
  render () {
    return (
      <div>
        <Input addTitle={this.addTitle.bind(this)} />
        <List data={this.state.list} />
        <List data={[1, 2, 3]} />
        <List data={['a', 'b', 'c']} />
        <List data={['x', 'y', 'z']} />
      </div>
    )
  }
  addTitle (title) {
    const currentList = this.state.list
    this.setState({
      list: currentList.concat(title)
    })
  }
}

export default Todo

class List extends Component {
  constructor (props) {
    super(props)
  }
  render () {
    const list = this.props.data
    return (
      <ul>
        {
          list.map((item, index) => {   // 复用
            return <li key={index}>{item}</li>
          })
        }
      </ul>
    )
  }
}

export default List

function render() {
  return React.createElement(
    'div',

    React.createElement(
      'p',
      null,
      'this is demo'
    ),
    React.createElement(Input, { addTitle: this.addTitle.bind(this) }),

  );
}

  1. 'div'可以直接渲染<div>,vdom可以做到

  2. Input和List,是自定义组件(class)，vdom默认不认识

  3. 因此，Input和List定义的时候必须声明render函数

  4. 根据props初始化实例，然后执行实例的render函数

  5. render函数返回的还是vnode对象

# setState---异步的

addTitle (title) {
  const currentList = this.state.list
  console.log(currentList);   // a, b
  this.setState({
    list: currentList.concat(title)  // c
  })
  this.setState({
    list: currentList.concat(title + 1)  
  })
  this.setState({
    list: currentList.concat(title + 2) // ---> c2
  })
  console.log(currentList);   // a, b
}
